86 * - Read encoder counter value from timer peripheral
87 * - Calculate velocity from change in position:

140void motor_update(void)
141{
142 /* First-order linear dynamics: speed approaches target_speed */
143 float target_speed = current_output * model_gain;
144 current_speed += model_alpha * (target_speed - current_speed);
145}
