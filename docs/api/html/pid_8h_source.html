22/**
23 * @brief PID Controller instance structure
24 *
25 * Contains all configuration and state for a single PID loop.
26 * Supports multiple independent instances. Do not modify members
27 * directly - use the API functions.
28 */
29typedef struct {
30 /* Configuration (set during initialization) */
31 float kp; /**< Proportional gain (0.1-10.0) */
32 float ki; /**< Integral gain (0.0-5.0, 0 to disable) */
33 float kd; /**< Derivative gain (0.0-1.0, 0 to disable) */
34 float dt; /**< Sample time in seconds (must match loop period) */
35 float out_min; /**< Minimum output limit */
36 float out_max; /**< Maximum output limit */
37 float integrator_min; /**< Min integrator limit (anti-windup) */
38 float integrator_max; /**< Max integrator limit (anti-windup) */
39 float derivative_lpf; /**< Derivative filter coeff (0.0-1.0, 0=no filter) */
40
41 /* Internal state (modified during operation) */
42 float integrator; /**< Integral accumulator */
43 float prev_error; /**< Previous error (for future use) */
44 float prev_measurement; /**< Previous measurement (for derivative) */
45 float derivative_filtered; /**< Filtered derivative value */
46} pid_t;

63 float ki,
64 float kd,

96/**
97 * @brief Calculate PID control output
98 *
99 * Must be called periodically at the rate specified by dt.
100 * Uses derivative-on-measurement to avoid derivative kick.
101 *
102 * @param pid Pointer to initialized PID structure
103 * @param setpoint Target value
104 * @param measurement Current measured value
105 * @return Control output clamped to [out_min, out_max]
106 */
107float pid_compute(pid_t *pid, float setpoint, float measurement);
