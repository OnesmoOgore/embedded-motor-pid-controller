32/* Target speed */

35int main(void)
36{
37 pid_t motor_pid;
38
39 /* Initialize motor and PID controller */
40 motor_init();
41 pid_init(&motor_pid, PID_KP, PID_KI, PID_KD, SAMPLE_TIME_S, OUT_MIN, OUT_MAX);
42
43 /* CSV header for simulation output */
44 printf("step,setpoint,measurement,output\n");
45
46 /* Control loop */
47 for (int step = 0; step < NUM_ITERATIONS; step++) {
48 /* Read current motor speed */
49 float measurement = motor_get_speed();
50
51 /* Compute PID control output */
52 float output = pid_compute(&motor_pid, SETPOINT, measurement);
53
54 /* Apply control output to motor */
55 motor_set_output(output);
56
57 /* Update motor simulation */
58 motor_update();
59
60 /* Log data (CSV format) */
61 printf("%d,%.4f,%.4f,%.4f\n", step, SETPOINT, measurement, output);
62 }
