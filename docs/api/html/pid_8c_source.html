25void pid_init(pid_t *pid,
26 float kp,
27 float ki,
28 float kd,
29 float dt,
30 float out_min,
31 float out_max)
32{
33 /* Validate inputs (assertions compile out in release builds) */
34 assert(pid != NULL && "PID structure pointer cannot be NULL");
35 assert(dt > 0.0f && "Sample time must be positive");
36 assert(kp >= 0.0f && "Proportional gain must be non-negative");
37 assert(ki >= 0.0f && "Integral gain must be non-negative");
38 assert(kd >= 0.0f && "Derivative gain must be non-negative");
39 assert(out_min < out_max && "Output min must be less than max");
40
41 /* Store configuration */
42 pid->kp = kp;
43 pid->ki = ki;
44 pid->kd = kd;
45 pid->dt = dt;
46 pid->out_min = out_min;
47 pid->out_max = out_max;
48
49 /* Initialize state to zero */
50 pid->integrator = 0.0f;
51 pid->prev_error = 0.0f;
52 pid->prev_measurement = 0.0f;
53 pid->derivative_filtered = 0.0f;
54
55 /* Calculate integrator limits (anti-windup) */
56 if (ki != 0.0f) {
57 pid->integrator_min = out_min / ki;
58 pid->integrator_max = out_max / ki;
59 } else {
60 pid->integrator_min = out_min;
61 pid->integrator_max = out_max;
62 }
63
64 /* No derivative filtering by default */
65 pid->derivative_lpf = 0.0f;
66}

110float pid_compute(pid_t *pid, float setpoint, float measurement)
111{
112 float error = setpoint - measurement;
113
114 /* Proportional term */
115 float p = pid->kp * error;
116
117 /* Integral term with anti-windup */
118 pid->integrator += error * pid->dt;
119 pid->integrator = clamp(pid->integrator, pid->integrator_min, pid->integrator_max);
120 float i = pid->ki * pid->integrator;
121
122 /* Derivative term (on measurement, not error)
123 * Negative sign: if measurement increases, we want negative D to oppose it.
124 * This avoids "derivative kick" when setpoint changes suddenly. */
125 float derivative_raw = -(measurement - pid->prev_measurement) / pid->dt;
126
127 /* Optional low-pass filter (exponential moving average) */
128 if (pid->derivative_lpf > 0.0f) {
129 pid->derivative_filtered = pid->derivative_filtered * pid->derivative_lpf +
130 derivative_raw * (1.0f - pid->derivative_lpf);
131 derivative_raw = pid->derivative_filtered;
132 }
133
134 float d = pid->kd * derivative_raw;
135
136 /* Combine and clamp output */
137 float output = p + i + d;
138 output = clamp(output, pid->out_min, pid->out_max);
139
140 /* Update state for next iteration */
141 pid->prev_error = error;
142 pid->prev_measurement = measurement;
143
144 return output;
145}
